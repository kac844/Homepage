int pin2 = 2; // letter O
int pin3 = 3; // letter L

int led;

void setup() {
  pinMode(pin1,OUTPUT);

}

void loop() {
    led = pin1;
    Ls();
    space();

    led = pin2;
    Lo();
    space();

    led = pin3;
    Ll();
    space();
    
    
}

void Lo(){
